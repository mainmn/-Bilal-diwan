تهذيب النفس

                        و ذكرتُ نفسي في الخلوات أفيقي  
                        و الدمع قبل ذكرها سابقٌ  
                        
                        ويح عيني إن خان ناظرها  
                        من يومٍ لسانها عليَّ ناطقُ  
                        
                        تتوارى عن الورى كأنما  
                        تحسبُ أن ما لها مرافقُ  
                        
                        و الشاهدان قد خطَّ قلمهما  
                        عبدًا للذنوبِ لا يفارقُ  
                        
                        قد أمسى على اللهو عاكفًا  
                        و من كأس الهوى هو ذائقُ  
                        
                        فيا ويلتي إذ الصحفُ طويت  
                        و جفتِ الآلامُ و صفتِ الخلائقُ
                        وجيء بالنفس قد أقبلت  
                        معها رفیقانِ، شهیدٌ و سائقُ  
                        
                        رب قد جئتك للغفرانِ ملتمسًا  
                        فأنا المخلوقُ وأنت الخالقُ  
                        
                        أنا عصيٌّ جُبلتُ على الخطايا  
                        أنت غفورٌ، للإصباحِ فالقُ  
                        
                        هب لي توبةً كنتُ أطلبها  
                        واجعلني بعبادك الصالحين لاحقُ  
                        
                        ما لي سبيلٌ إليك إلا الرجا  
                        فعبدُك منيبٌ و قلبُه صادقُ

أديبٌ في شعره مللُ

                        قــالــت أديــبٌ و فـي شِــعــره مــلــلُ
                        مـا بــدى مـن قـولـه الـهـوى و الـغــزلُ
                        
                        جــافٍ لا يــبــتــغِ فــي وَصْــلِـه مــودَّةً
                        أم عَــفِــيـفٌ سِـمـاهُ الـحَـيـا و الـخَـجـلُ
                        
                        قلـت يا ربَّة الحُسن يا من عـنـي تسـألُ
                        فـي وصـفـكِ تاهـت القـوافي و الجـمـلُ
                        
                        قـد كُــنـتُ أنـسـجُ مـن الـحـروف غــزلاً
                        و عِــنــد لُـقــيـاكِ  غــادرتــنــي الـحِـيـلُ
                        
                        يـا وردةً فـاحَ فـي الــرَّوضِ عــبــيـرُهـا
                        يـا نــحــلــةً تـلالا مــن فــمِــها الـعـسـلُ
                        
                        عــيــنـاكِ الــســلامُ فـي حــربٍ ثــائـرةٍ
                        عـيــنــاكِ لـلــيــائِــس الــحــزِيــنِ أمـــلُ
                        
                        عــيــنــاكِ نـبــيــذٌ فــاضَــتْ كــؤُوسُــها
                        الـــفُــــؤادُ صــريــعٌ و الــعــقْـــلُ ثــمِــلُ
                        
                        قـد سـألـتُ قـلـبـًا مــتَـيَّـمـا عــن دوائـهِ
                        قَــال بــســمــاتُــها  فــإنــنــي مُــثــقــلُ
                        
                        بـهـا تــطــيــبُ جُــروح الــهُـيـامِ و بـهـا
                        تـنــجـلــي عـنـي الأســقَــامُ و الــعِــلَــلُ
                        
                        هــي الـــرُّوح و بــالـــرُّوحِ فـــديـــتــهـا
                        مـــا فـــارقــنـــي عـــنـــهـــا إلا الأجـــلُ
                        
                        يـا ربَّـة الـحُـسـن فـاسـمـعي مـنـه قـولاً
                        قـائـلـهُ صــدوقٌ لا يــعــتـلـيـه الــدَّجـلُ
                        
                        قــالــت أديـــبٌ و فــي شِــعــره مــلــلُ
                        قُــلــتُ شِــعــري سَـلـيــمٌ مـا بــه خـلـلُ
                        
                        تـلـك الـحـروف أفـصـحت فـي مــقالـها
                        أنــكِ خــيــرُ مـا نــظــرت إلـيـه الـمُـقـلُ
                        
                        يا ربـة الحُـسـن قصـيدةً أنـتِ عنـوانـها
                        حـسـنـاءٌ مــالــها فــي الــوجــود مَــثـلُ

لغة الضاد

        و مهما شـرِب المرءُ من ضـادٍ
        زادَ ضـــمـــأُهُ كُــلـمــا شَــرِبَــا
        
        فراحَ يــغُــوص فــي أعماقـها
        و كُــلَّــما غــاص قــد عَـجِــبـا
        
        مَــلِــكـةُ الـلُّـغـاتِ بــحــرٌ هِــيِ
        و اللِـسـانُ مـن نـطــقـها طَـرِبَا
        
        فـأمـسـى فصـيـحَ القـولِ بيِّـنًا
        و ألجم من ظنه لبِيدًا و حَسِبَا
        
        هـم أعـرابٌ للـعـروبـةِ نُـسـبـوا
        و أنـا عـربـيٌ للـعـربـيـةِ نُـسِـبَـا
        
        فـلا شِـعــري يـفـي بــمـدحـهـا
        و لــو بــالـذهــبٍ قــد كُــتِــبَـا

حديث القمر

        
الأرقُ جليـسي و طـال بنا السـمر
رنوتُ نحو السَّـمـا فلمـحتُ القـمر 

قـلـتُ يـا بدرًا أضـاء الدجى نورهُ
أ إنـك وحـيـدٌ أم تُـحـب الـسـهـر؟

قـال لــبَّـــيـكَ جـــوابِـي فـإنــنــي
قـد لــقِــيـتُ نَـصَـبِـي من الـسَّـفـر

أوكُلما قدِمتُ مع الشفقِ مُـرتـحلاً
غــابــت شــمــسـي عــن الـنــظـر

أفــلــت فــمــا أدركــتُ نُـــورهـــا
و مـا بـان لــهـا فـي الــكــون أثـر

فلا النـجم ولا الشِّـهـابُ يؤنـسنـي
ولا الكـواكـبُ ولا نـظـراتُ البـشـر

و هــذا بـــيــانٌ مـــن ربِّ الـــقــدر
لا الشمس ينبغي لها أن تدرك القمر
